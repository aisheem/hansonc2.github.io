import markdown import os

def convert_all() -> None: ''' Convert all markdown files in current directory to html ''' for name in os.listdir(): with open(name, 'r') as f: text = f.read() html = markdown.markdown(text)

    with open('html/' + name[:-3] + '.html', 'w') as f:
        f.write(html)

def main(): convert_all()

if name == 'main': main()
